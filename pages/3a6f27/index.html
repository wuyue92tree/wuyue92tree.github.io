              flume采集数据到hdfs

              # 简单示例 helloworld

              # flume配置文件

              # angent名称为`demo`

# sources/channels/sinks分别为r1/c1/k1
demo.sources  = r1
demo.channels = c1
demo.sinks    = k1

# 以netcat为例，监听socket接口`9999`输入
demo.sources.r1.type     = netcat
demo.sources.r1.bind     = 127.0.0.1
demo.sources.r1.port     = 9999

demo.channels.c1.type = memory
demo.channels.c1.capacity = 1000
demo.channels.c1.transactionCapacity = 100

# 将采集到的日志直接输出到flume日志中
demo.sinks.k1.type         = logger


demo.sources.r1.channels   = c1
demo.sinks.k1.channel      = c1

              # flume输出到hdfs示例

              以nginx日志为例，将nginx日志通过flume采集到hdfs中

              # 一级flume配置文件

              # angent名称为`demo_hdfs`

# sources/channels/sinks分别为r1/c1/k1
demo_hdfs.sources  = r1
demo_hdfs.channels = c1
demo_hdfs.sinks    = k1

demo_hdfs.sources.r1.type = TAILDIR
demo_hdfs.sources.r1.positionFile     = /bigdata/flume/taildir_position.json
demo_hdfs.sources.r1.filegroups     = f1 f2
demo_hdfs.sources.r1.filegroups.f1 = /var/log/nginx/access.log

demo_hdfs.channels.c1.type = memory
demo_hdfs.channels.c1.capacity = 1000
demo_hdfs.channels.c1.transactionCapacity = 100

# 将采集到的日志直接输出到`二级flume`中
demo_hdfs.sinks.k1.type         = avro
demo_hdfs.sinks.k1.hostname = 10.0.4.60
demo_hdfs.sinks.k1.port = 54321


demo_hdfs.sources.r1.channels   = c1
demo_hdfs.sinks.k1.channel      = c1

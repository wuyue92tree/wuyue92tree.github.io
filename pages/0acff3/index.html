将console.log日志输出到页面

# vue.js 代码
# 1. 在mounted中初始化loggerMan
# 2. 在data中定义 {log_list: []}
# 使用console.log时会将日志unshift到log_list，在页面展示log_list即可

loggerMan () {
var old = console.log
// var logger = document.getElementById('log')
let self = this
console.log = function (message) {
if (typeof message === 'object') {
message = JSON && JSON.stringify ? JSON.stringify(message) : message
}
if (message.indexOf('采集日志') !== -1) {
self.log_list.unshift(chalk.blue(message))
}
if (self.log_list.length === 0) {
self.log_list.unshift(chalk.blue('还没有产生采集日志.'))
}
old(chalk.blue(message))
}
}

js模拟键盘输入

function DoHelp(el, character) {
var e = new KeyboardEvent("keydown", { bubbles: true, cancelable: true, key: character, char: character, shiftKey: true, keyCode: character.charCodeAt(0) });
el.dispatchEvent(e)
}
